### Responsive Column Logic

<CodeGroup>

```javascript Dynamic Columns
function chunkArray(array, size) {
  if (!Array.isArray(array)) return [];
  const result = [];
  for (let i = 0; i < array.length; i += size) {
    result.push(array.slice(i, i + size));
  }
  return result;
}

const [chunkedSkills, setChunkedSkills] = useState([]);

useEffect(() => {
  const updateCols = () => {
    const width = window.innerWidth;
    const cols = width >= 1024 ? 6 : width >= 768 ? 4 : 2;
    setChunkedSkills(chunkArray(skills, cols));
  };

  updateCols();
  window.addEventListener('resize', updateCols);
  return () => window.removeEventListener('resize', updateCols);
}, []);
```

</CodeGroup>

### Scroll Animation Implementation

<AccordionGroup>
  <Accordion title="Scroll Progress Tracking">
    ```javascript
    const ref = useRef(null);
    const containerRef = useRef(null);
    const [height, setHeight] = useState(0);

    const { scrollYProgress } = useScroll({
      target: containerRef,
      offset: ['start 10%', 'end 50%'],
    });

    const heightTransform = useTransform(scrollYProgress, [0, 1], [0, height]);
    const opacityTransform = useTransform(scrollYProgress, [0, 0.1], [0, 1]);
    ```
    
    The scroll progress is tracked from when the container enters the viewport (start 10%) until it's halfway out (end 50%).
  </Accordion>

  <Accordion title="Animated Progress Line">
    ```jsx
    <motion.div
      style={{
        height: heightTransform,
        opacity: opacityTransform,
      }}
      className="absolute inset-x-0 top-0 w-[2px] rounded-full bg-gradient-to-t from-[var(--accent)] from-[0%] via-[var(--main)] via-[10%] to-transparent"
    />
    ```
    
    The line fills from bottom to top as you scroll, using a gradient from accent color to transparent.
  </Accordion>
</AccordionGroup>

      <div
        className="mb-5 text-[14px] leading-[24px] md:text-[16px]"
        dangerouslySetInnerHTML={{ __html: item.textContent }}
      />
    )}
    
    {/* Images */}
    {item.imgContent && (
      <div className={item.imgWrapperCss}>
        {item.imgContent.map((src, i) => (
          <Image
            key={i}
            src={src}
            alt={`Timeline image ${i + 1}`}
            width={0}
            height={0}
            className={`w-full rounded-[10px] border-[1px] border-[var(--border-default)] object-cover ${item.imgContentCss}`}
          />
        ))}
      </div>
    )}
  </div>
</div>

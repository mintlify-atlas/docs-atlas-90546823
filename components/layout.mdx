## Navbar

The main navigation component with responsive mobile menu and smooth transitions.

**Location:** `src/components/Layout/Navbar/Navbar.js`

### Features

- Sticky positioning with backdrop blur
- Responsive mobile dropdown menu
- Smooth animation transitions
- Logo-based branding
- Lucide React icons integration

### Implementation

<CodeGroup>

```javascript Navbar.js
'use client';

import React, { useState } from 'react';
import Link from 'next/link';
import { ChevronDown, ChevronUp, UserRound, Code, Mail, FileText } from 'lucide-react';
import NavbarLink from './NavbarLinks';

export default function Navbar() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className="sticky top-0 z-50 m-auto max-w-[980px] md:top-5">
      <nav className="text-md border-b-[1px] border-b-[var(--border-default)] bg-[var(--surface)] px-5 py-3 backdrop-blur-sm md:rounded-full md:border-[1px]">
        <div className="flex items-center justify-between">
          <Link href="/" className="text-lg font-semibold">
            <img src="/arsh-favicon.png" className="w-[35px] md:w-[50px]" />
          </Link>

          {/* Mobile toggle */}
          <button
            className="rounded-full border border-[var(--accent)] bg-[var(--main)] p-[0.5px] text-[#f5f5f7] md:hidden"
            onClick={() => setIsOpen(!isOpen)}
          >
            {isOpen ? <ChevronUp className="h-5 w-5" /> : <ChevronDown className="h-5 w-5" />}
          </button>

          {/* Desktop links */}
          <div className="hidden md:flex md:pr-3">
            <NavbarLink href="/about" icon={UserRound} label="About" />
            <NavbarLink href="/resume" icon={FileText} label="Resume" />
            <NavbarLink href="/made-by-arsh" icon={Code} label="Made by Arsh" />
            <NavbarLink href="/connect" icon={Mail} label="Connect" />
          </div>
        </div>

        {/* Mobile links dropdown */}
        <div
          className={`overflow-hidden transition-all duration-500 md:hidden ${
            isOpen ? 'max-h-50 mt-4 border-t border-[var(--border-default)]' : 'max-h-0'
          }`}
        >
          <div className="flex flex-col space-y-4 py-5 pb-2">
            <NavbarLink href="/about" icon={UserRound} label="About" onClick={() => setIsOpen(false)} />
            <NavbarLink href="/resume" icon={FileText} label="Resume" onClick={() => setIsOpen(false)} />
            <NavbarLink href="/made-by-arsh" icon={Code} label="Made by Arsh" onClick={() => setIsOpen(false)} />
            <NavbarLink href="/connect" icon={Mail} label="Connect" onClick={() => setIsOpen(false)} />
          </div>
        </div>
      </nav>
    </div>
  );
}
```

</CodeGroup>

### Key Design Patterns

<AccordionGroup>
  <Accordion title="Responsive Behavior">
    - Desktop: Horizontal navigation with rounded pill design
    - Mobile: Collapsible dropdown menu with smooth height animation
    - Breakpoint: `md` (768px)
  </Accordion>

  <Accordion title="Sticky Positioning">
    The navbar uses `sticky top-0` on mobile and `sticky top-5` on desktop to create a floating effect while scrolling.
  </Accordion>

  <Accordion title="CSS Variables">
    - `--border-default`: Border color
    - `--surface`: Background color
    - `--accent`: Accent color for buttons
    - `--main`: Primary color for interactive elements
  </Accordion>
</AccordionGroup>

## NavbarLink

Reusable navigation link component with icon and hover effects.

**Location:** `src/components/Layout/Navbar/NavbarLinks.js`

### Props

<ParamField path="href" type="string" required>
  The destination URL for the link
</ParamField>

<ParamField path="icon" type="LucideIcon" required>
  Lucide React icon component to display
</ParamField>

<ParamField path="label" type="string" required>
  Text label for the navigation item
</ParamField>

<ParamField path="onClick" type="function">
  Optional click handler (used to close mobile menu)
</ParamField>

### Implementation

```javascript NavbarLinks.js
'use client';

import Link from 'next/link';

export default function NavbarLink({ href, icon: Icon, label, onClick }) {
  return (
    <Link href={href} onClick={onClick} className="group/link">
      <div className="group flex w-fit items-center rounded-[25px] align-middle hover:text-[var(--accent)] md:px-3 md:py-1 md:hover:bg-[var(--main)] md:hover:text-[#f5f5f7]">
        <Icon className="mr-4 w-[15px] pb-[1px] md:mr-1.5" />
        {label}
      </div>
    </Link>
  );
}
```

### Usage Example

```jsx
import { UserRound } from 'lucide-react';
import NavbarLink from '@/components/Layout/Navbar/NavbarLinks';

<NavbarLink 
  href="/about" 
  icon={UserRound} 
  label="About"
  onClick={() => console.log('Clicked')}
/>
```

### Time Display Logic

```javascript Time Formatting
const showTime = () => {
  const time = new Date();
  const month = time.toLocaleString('default', { month: 'long' });
  const date = time.getDate();
  const year = time.getFullYear();
  let hour = time.getHours();
  let min = time.getMinutes();
  let sec = time.getSeconds();
  let am_pm = 'AM';

  if (hour > 12) {
    hour -= 12;
    am_pm = 'PM';
  } else if (hour === 0) {
    hour = 12;
  }

  const formattedTime = `${month} ${date}, ${year} • ${hour < 10 ? '0' + hour : hour}:${min < 10 ? '0' + min : min}:${sec < 10 ? '0' + sec : sec} ${am_pm}`;
  
  setTimeString(formattedTime);
};
```

## TopButton

Scroll-to-top button with smooth scrolling animation.

**Location:** `src/components/Layout/Footer/TopButton.js`

### Implementation

```javascript TopButton.js
'use client';

import { ChevronUp } from 'lucide-react';

export default function ScrollToTop() {
  const scrollToTop = () => {
    window.scrollTo({ top: 0, behavior: 'smooth' });
  };

  return (
    <button
      onClick={scrollToTop}
      title="Go to top"
      className="z-50 ml-2 rounded-full border-[1px] border-[var(--main)] px-[1px] text-[var(--foreground)] hover:bg-[var(--main)] hover:text-[#f5f5f7]"
    >
      <ChevronUp className="h-5 w-5" />
    </button>
  );
}
```

### Usage

The TopButton is automatically included in the Footer component alongside the ThemeToggle.

<Note>
  The `behavior: 'smooth'` property provides a smooth scrolling animation instead of an instant jump.
</Note>
